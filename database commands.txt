create table ENQUIRY(enroll_id int primary key,name varchar(100),inquired_course varchar(100),age int,qualification varchar(100),contact long,email varchar(100),gender varchar(100));


create table follow_up(enroll_id int,no_of_attempt int,foreign key(enroll_id) references enquiry (enroll_id));


create table course(name varchar(100) primary key,fees int,duration varchar(100));


create table batch(batch_id int primary key,name varchar(100),seat int,batch_timing varchar(100),FOREIGN KEY (name) REFERENCES course(name));


create table Admitted_student(id int primary key,name varchar(100),age int,qualification varchar(100),cname varchar(100),fees int,gender varchar(100),batch_timing varchar(100),batch_id int,FOREIGN KEY (batch_id) REFERENCES batch(batch_id),FOREIGN KEY (cname) REFERENCES course(name));


create table faculty(name varchar(100) primary key,experience varchar(100),specialization varchar(100),education varchar(100),photo_link varchar(100));


create table admin(name varchar(100) primary key,password varchar(100));
